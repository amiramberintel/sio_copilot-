=== SETUP ===

  # Open/create work area
  /nfs/site/proj/hdk/pu_tu/prd/liteinfra/1.19.p1/commonFlow/bin/cth_psetup \
    -proj gfc_n2_client/GFC_TS2025.17 -nowash -cfg gfcn2clienta0.cth \
    -ward <WA_PATH> -x '$SETUP_R2G'

  # Reopen existing WA
  source $WA/sd.reopen

  # Copy scripts from reference
  cp -rf $REF_WA/fcn_scripts/ .

  # Change reference WA
  rm ref && ln -s <new_ref_wa> ref

=== RUN FLOW ===

  # Prepare + run synthesis
  fcn_prepare -run

  # Monitor jobs
  fcn_status -wait

  # Mid-run checkers
  fcn_cdns_check_genus_map          # All partitions finished Genus?
  fcn_snps_check_init_floorplan     # All partitions finished FC?

  # Generate FDRs
  fcn_prepare -run_fdr

  # Promote (no checks, with tag)
  fcn_promote -no_check -tag <TAG>

  # Send summary email
  fcn_mail_summary

  # Run FEV
  fcn_promote -no_check -run_fev

  # Debug FEV
  debug_fev $ward/runs/core_client/1278.6/fev_conformal/fev_fcl/checkpt_flat.lec

  # Commit to archive
  $ward/fcn_commit.csh

=== FULL ONE-LINER ===

  fcn_prepare -run ; \
  self_mail fcn_GFC_client_started ; \
  fcn_status -wait ; \
  self_mail fcn_client_finished ; \
  fcn_prepare -run_fdr ; \
  fcn_status -wait ; \
  self_mail fcn_client_FDR_finished ; \
  fcn_scripts/fcn_check_runs.tcl ; \
  fcn_promote -no_check -tag <YOUR_TAG> ; \
  fcn_mail_summary ; \
  fcn_promote -no_check -run_fev ; \
  fcn_status -wait ; \
  self_mail fcn_client_FEV_finish

  # Then after review:
  $ward/fcn_commit.csh

=== CHECKS ===

  # Pre-commit error check (must be clean)
  grep -- "-E-" fcn_promote.log
  grep -- "-W-" fcn_promote.log
  grep -- "-E-" fcn_reports/run_check_latest/*
  grep -- "-W-" fcn_reports/run_check_latest/*
  grep -- "-E-" fcn_reports/csv_latest/*

  # View indicators
  cat $WA/fcn_reports/csv_latest/fdr_cnt.compare.csv
  cat $WA/fcn_reports/csv_latest/seq_and_cell.compare.csv
  cat $WA/fcn_reports/csv_latest/hips_cnt.compare.csv

  # View partition logs
  less fcn_logs/job.<par>.cdns.log
  less fcn_logs/job.<par>.snps.log
  less fcn_logs/job.<par>.fdr.log

=== BLACK BOX / HIP TOOLS ===

  # Compare HIPs XML vs RTL
  fcn_xml_hips_vs_rtl_diff <HIP_TAG> <RTL_MODEL>

  # Extract BB list from comparison
  fcn_xml_hips_vs_rtl_diff <HIP_TAG> <RTL_MODEL> \
    | l > fcn_xml_hips_vs_rtl_diff_output.rpt
  l fcn_xml_hips_vs_rtl_diff_output.rpt \
    | g "EBB not in XML" | sed 's/EBB not in XML://' \
    | sort | sed 's/$/\\/' | l

  # Generate ports diff (TST vs REF)
  fcn_model_comp_tst_vs_ref_client
